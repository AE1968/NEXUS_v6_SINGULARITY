function updateStatusDisplay(isLoggedIn, username = '', subscription = '') {

// Demo countdown
let demoCountdownInterval = null;

function startDemoCountdown(username, gender, subscription) {
    const loginTime = new Date(localStorage.getItem('nexus_login_time'));
    const endTime = new Date(loginTime.getTime() + 10 * 60 * 1000); // +10 min
    
    // Clear previous interval
    if (demoCountdownInterval) {
        clearInterval(demoCountdownInterval);
    }
    
    demoCountdownInterval = setInterval(() => {
        const now = new Date();
        const timeLeft = endTime - now;
        
        if (timeLeft <= 0) {
            clearInterval(demoCountdownInterval);
            alert('Demo session expired (10 minutes)');
            window.logout();
            return;
        }
        
        const minutes = Math.floor(timeLeft / 60000);
        const seconds = Math.floor((timeLeft % 60000) / 1000);
        const timeString = `${minutes}:${seconds.toString().padStart(2, '0')}`;
        
        const statusDisplay = document.getElementById('status-display');
        if (statusDisplay) {
            statusDisplay.innerHTML = `${username.toUpperCase()} [${subscription.toUpperCase()}] | ${gender.toUpperCase()} | <span id="system-status">READY (${timeString} left)</span>`;
        }
    }, 1000); // Update every second
}

// Get days until expiry
function getDaysUntilExpiry(username) {
    const user = USERS_DB[username];
    if (!user || !user.subscriptionDetails || !user.subscriptionDetails.endDate) return 0;
    
    const now = new Date();
    const endDate = new Date(user.subscriptionDetails.endDate);
    return Math.ceil((endDate - now) / (1000 * 60 * 60 * 24));
}
