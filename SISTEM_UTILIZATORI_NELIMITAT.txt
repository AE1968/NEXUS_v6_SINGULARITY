function validateRegistration(username, email, password) {
    // 1. Username unic
    if (USERS_DB[username]) {
        return { valid: false, error: 'Username already exists' };
    }
    
    // 2. Username format
    if (!/^[a-zA-Z0-9_]{3,20}$/.test(username)) {
        return { valid: false, error: 'Invalid username format' };
    }
    
    // 3. Email valid
    if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)) {
        return { valid: false, error: 'Invalid email format' };
    }
    
    // 4. Email unic
    for (let user in USERS_DB) {
        if (USERS_DB[user].email === email) {
            return { valid: false, error: 'Email already registered' };
        }
    }
    
    // 5. Password strength
    if (password.length < 8) {
        return { valid: false, error: 'Password too short (min 8 chars)' };
    }
    
    return { valid: true };
}

            stats.registeredUsers++;
